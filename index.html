How to use Grunt

$ npm install grunt-cli -g
$ mkdir project && cd project

$ npm init
$ npm install grunt --save-dev

<Define Gruntfile.{js,coffee}>
$ grunt <task>

Gruntfile.{js,coffee}

module.exports = function(grunt) {
  grunt.initConfig({
    concat: {
      dist: {
        src: ['.tmp/js/*.js'],
        dest: 'js/app.js'
      }
    },
    uglify: {
      dist: {
        src: 'js/app.js',
        dest: 'js/app.min.js'
      }
    }
  });
};
